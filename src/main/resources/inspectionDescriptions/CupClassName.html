Reports class names that may cause conflicts. 

    // Class name can't have the same name as production if production has more than one items on the right-hand side.
    ExampleA ::= (ExampleA) Item | Item ;
    ExampleAFine ::= (ExampleAFine) Item ; // This will not cause a conflict.

    // Class name can't have the same name as different productions.
    ExampleB1 ::= (OtherName) Item ;
    ExampleB2 ::= (ExampleB1) Item ;

    // Class names are unique for each item.
    ExampleC1 ::= (SameName) Item | (SameName) Item ;
    ExampleC2 ::= (SameName) Item ;
